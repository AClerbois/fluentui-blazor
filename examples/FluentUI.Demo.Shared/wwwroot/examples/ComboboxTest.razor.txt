
<h5>From list of Option&lt;string&gt; items</h5>
<p>Second item in the list is initially selected , therefore a <code>OptionSelected</code> parameter is added</p>
<FluentCombobox TOption="Option<string>" Items=@stringOptions 
                OptionText="@(i => i.Text)" 
                OptionValue="@(i => i.Value)" 
                OptionSelected=@(i => i.Selected)
                @bind-Value="@comboboxStringListValue" 
                @bind-SelectedItem=@selectedStringListOption />
<p>
    Selected Value: @comboboxStringListValue<br />
    Selected Item (strongly typed): <br />
    Value: @selectedStringListOption?.Value (Type: @(selectedStringListOption?.Value.GetType())) <br />
    Text: @selectedStringListOption?.Text
</p>

<h5>From list of Option&lt;int&gt; items</h5>
<p>First item in the list is disabled, therefore a <code>OptionDisabled</code> parameter is added</p>
<FluentCombobox Items=@intOptions 
            OptionText="@(i => i.Text.ToString())" 
            OptionValue="@(i => i.Value.ToString())" 
            OptionDisabled=@(i => i.Disabled)
            @bind-SelectedItem="@selectedIntListOption"
                @bind-Value="@comboboxIntListValue" />
<p>
    Selected Value: @comboboxIntListValue<br />
    Selected Item (strongly typed): <br />
    Value: @selectedIntListOption?.Value (Type: @(selectedIntListOption?.Value.GetType())) <br />
    Text: @selectedIntListOption?.Text
</p>


@code {

    string? comboboxStringListValue;
    Option<string>? selectedStringListOption;

    string? comboboxIntListValue;
    Option<int>? selectedIntListOption;

    List<Option<string>> stringOptions = new()
        {
            { new Option<string> { Value = "1", Text = "One" } },
            { new Option<string> { Value = "2", Text = "Two", Selected = true } },
            { new Option<string> { Value = "3", Text = "Three" } }

        };

    List<Option<int>> intOptions = new()
        {
            { new Option<int> { Value = 1, Text = 1, Disabled = true } },
            { new Option<int> { Value = 2, Text = 2 } },
            { new Option<int> { Value = 3, Text = 3 } }

        };


}
